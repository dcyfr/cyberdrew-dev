## Code blocks (TypeScript)

```ts
type User = {
	id: string
	name: string
	roles: ("admin" | "editor" | "viewer")[]
}

export function greet(user: User) {
	const role = user.roles[0] ?? "viewer"
	return `Hello, ${user.name}! Your role is ${role}.`
}

console.log(greet({ id: "1", name: "Ada", roles: ["admin"] }))
```
